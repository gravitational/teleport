# Web UI Audit Event Generation — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Auto-generate a TypeScript file containing audit event metadata (codes, types, formatters, icons, fixtures) so new resource-gen resources appear in the web UI audit log with zero manual TS work.

**Architecture:** A new gathering generator (`web_events_gathering.go`) reuses the existing `buildEventEntries()` helper to collect event metadata from all specs, then renders a TypeScript template (`web_events_ts.ts.tmpl`) into `web/packages/teleport/src/services/audit/generatedResourceEvents.gen.ts`. A new `--web-dir` CLI flag controls output. Four existing TS files get one-time wiring to import and spread the generated values.

**Tech Stack:** Go (generator), Go text/template (rendering), TypeScript (output)

### Task 2: Create the TypeScript template

**Step 1: Write the template**

The template receives a `webEventsTSData` struct with an `Events` field (slice of `webEventTSEntry`). Each entry has: `UpperSnakeName`, `Code`, `EventType`, `DisplayName`, `Article`, `PastVerb`, `Lower`, `OpLower`, `FixtureUID`.

```
// Code generated by resource-gen; DO NOT EDIT.

import type { RawEvent } from './types';

type HasName = { name: string };

// Event codes for resource-gen managed resources.
export const generatedEventCodes = {
{{- range .Events}}
  {{.UpperSnakeName}}: '{{.Code}}',
{{- end}}
} as const;

export type GeneratedEventCode =
  (typeof generatedEventCodes)[keyof typeof generatedEventCodes];

// RawEvent type entries for generated resource events.
export type GeneratedRawEvents = {
{{- range .Events}}
  [typeof generatedEventCodes.{{.UpperSnakeName}}]: RawEvent<
    typeof generatedEventCodes.{{.UpperSnakeName}},
    HasName
  >;
{{- end}}
};

// Formatter entries for generated resource events.
export function generatedFormatters() {
  return {
{{- range .Events}}
    [generatedEventCodes.{{.UpperSnakeName}}]: {
      type: '{{.EventType}}',
      desc: '{{.DisplayName}} {{.PastVerb | title}}',
      format: ({ user, name }: { user: string; name: string }) =>
        ` + "`" + `User [${user}] {{.PastVerb}} {{.Article}} {{.DisplayName | lower}} [${name}]` + "`" + `,
    },
{{- end}}
  };
}

// Icon mappings for generated resource events.
export const generatedEventIcons: Record<GeneratedEventCode, string> = {
{{- range .Events}}
  [generatedEventCodes.{{.UpperSnakeName}}]: 'Info',
{{- end}}
};

// Fixture entries for generated resource events.
export const generatedFixtures = [
{{- range .Events}}
  {
    code: '{{.Code}}' as const,
    event: '{{.EventType}}',
    name: 'example',
    user: 'alice',
    time: '2026-01-01T00:00:00Z',
    uid: '{{.FixtureUID}}',
  },
{{- end}}
];
```

```go
package generators

import (
	"fmt"
	"strings"
	"unicode"

	"github.com/gravitational/teleport/build.assets/tooling/cmd/resource-gen/spec"
	"github.com/gravitational/trace"
)

// webEventTSEntry holds template data for a single event in the TS output.
type webEventTSEntry struct {
	UpperSnakeName string // ACCESS_POLICY_CREATE
	Code           string // AP001I
	EventType      string // resource.access_policy.create
	DisplayName    string // Access Policy
	Article        string // a or an
	PastVerb       string // created, updated, deleted, read
	Lower          string // access_policy
	OpLower        string // create
	FixtureUID     string // deterministic UUID placeholder
}

type webEventsTSData struct {
	Events []webEventTSEntry
}

var webEventsTSTmpl = mustReadTemplate("web_events_ts.ts.tmpl")

// GenerateWebEventsTS renders the TypeScript file for web UI audit events.
func GenerateWebEventsTS(specs []spec.ResourceSpec) (string, error) {
	entries := buildWebEventEntries(specs)
	data := webEventsTSData{Events: entries}
	out, err := renderTS("webEventsTS", webEventsTSTmpl, data)
	if err != nil {
		return "", trace.Wrap(err)
	}
	return out, nil
}

func buildWebEventEntries(specs []spec.ResourceSpec) []webEventTSEntry {
	goEntries := buildEventEntries(specs)
	tsEntries := make([]webEventTSEntry, 0, len(goEntries))
	// Build a map from kind to KindPascal for display name lookup.
	pascalByKind := map[string]string{}
	for _, rs := range specs {
		pascalByKind[rs.Kind] = rs.KindPascal
	}
	for i, e := range goEntries {
		pascal := pascalByKind[e.Lower]
		display := pascalToDisplayName(pascal)
		tsEntries = append(tsEntries, webEventTSEntry{
			UpperSnakeName: pascalToUpperSnake(e.ConstName),
			Code:           e.Code,
			EventType:      fmt.Sprintf("resource.%s.%s", e.Lower, e.OpLower),
			DisplayName:    display,
			Article:        article(display),
			PastVerb:       pastVerb(e.OpLower),
			Lower:          e.Lower,
			OpLower:        e.OpLower,
			FixtureUID:     fmt.Sprintf("00000000-0000-0000-0000-%012d", i+1),
		})
	}
	return tsEntries
}

// pascalToUpperSnake converts "AccessPolicy" to "ACCESS_POLICY".
func pascalToUpperSnake(s string) string {
	var result strings.Builder
	for i, r := range s {
		if unicode.IsUpper(r) && i > 0 {
			prev := rune(s[i-1])
			if unicode.IsLower(prev) {
				result.WriteRune('_')
			}
		}
		result.WriteRune(unicode.ToUpper(r))
	}
	return result.String()
}

// pascalToDisplayName converts "AccessPolicy" to "Access Policy".
func pascalToDisplayName(s string) string {
	var result strings.Builder
	for i, r := range s {
		if unicode.IsUpper(r) && i > 0 {
			prev := rune(s[i-1])
			if unicode.IsLower(prev) {
				result.WriteRune(' ')
			}
		}
		result.WriteRune(r)
	}
	return result.String()
}

// article returns "an" if the display name starts with a vowel, "a" otherwise.
func article(displayName string) string {
	if len(displayName) == 0 {
		return "a"
	}
	switch unicode.ToLower(rune(displayName[0])) {
	case 'a', 'e', 'i', 'o', 'u':
		return "an"
	default:
		return "a"
	}
}

// pastVerb returns the past tense of a CRUD operation.
func pastVerb(op string) string {
	switch op {
	case "create":
		return "created"
	case "update":
		return "updated"
	case "delete":
		return "deleted"
	case "get":
		return "read"
	default:
		return op + "d"
	}
}

// renderTS is like render but adds custom template functions for TS output.
func renderTS(name string, tmpl string, data any) (string, error) {
	t, err := template.New(name).Funcs(template.FuncMap{
		"lower": strings.ToLower,
		"title": func(s string) string {
			if len(s) == 0 {
				return s
			}
			return strings.ToUpper(s[:1]) + s[1:]
		},
	}).Parse(tmpl)
	if err != nil {
		return "", trace.Wrap(err)
	}
	var b strings.Builder
	if err := t.Execute(&b, data); err != nil {
		return "", trace.Wrap(err)
	}
	return b.String(), nil
}
```
